{{- /* If it is part of a series, link to related articles */ -}} {{- $seeAlso := (slice) -}} {{- $permalink := .Permalink -}} {{- $siteSeries := .Site.Taxonomies.series -}} {{- with .Params.series -}} {{- range $name := . -}} {{- $series := index $siteSeries ($name | urlize) -}} {{- range $page := first 6 $series.Pages -}} {{- if ne $page.Permalink $permalink -}} {{- $seeAlso = $seeAlso | append $page.Permalink -}} {{- end -}} {{- end -}} {{- end -}} {{- end -}} {{- /* Otherwise, use Hugo's related content function */ -}} {{- if le (len $seeAlso) 0 -}} {{- $related := .Site.Pages.Related . | first 6 -}} {{- with $related -}} {{- range . -}} {{- $seeAlso = $seeAlso | append .Permalink -}} {{- end -}} {{- end -}} {{- end -}} {{- return $seeAlso -}}
